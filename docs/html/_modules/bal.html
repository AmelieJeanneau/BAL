[docs]def reclass_veg(veg, dem):
    """ 
    Reclassify the original vegetation into the categories classified as table
    2.3 in AS 3959 (2009).
    
    :param veg: `file` the input vegetation
    :param dem: `file` the input dem used to get the area of interest (AOI)
    
    :return: `file` the reclassified vegetation 
    """ 
    
    input_folder = os.path.dirname(veg) 
    parent = os.path.abspath(pjoin(input_folder, os.pardir))
    output_folder = pjoin(parent, 'output')   

    arcpy.env.workspace = input_folder

    veg_r = 'big_area_r'
    veg_r_proj = 'cls_r_pj'
    veg_class_r = output_folder + '\\class_r'
    
    remap = "1 3;2 2;3 3;4 7;5 7;6 1;7 1;8 1;9 2;10 3;11 1;12 1;13 1;14 3;\
            15 6;16 7;17 3;18 2;19 2;20 7;21 3;22 2; 23 7;24 7;25 5;26 5;27 5;\
            28 2;29 7;30 7;31 3;32 1;33 1;34 7;35 3;36 7;37 7;38 4;39 7;40 1;\
            41 2;42 7;43 2;44 7;45 4;46 7;47 NODATA;97 2;98 1;99 1"

    log.info('remap the vegetation ' + remap)    
    
    # Derive reclassifed veg...
    Reclassify(veg, "Value", remap, "NODATA").save(veg_r)
    
    # project as dem and change cell size as 25
    arcpy.ProjectRaster_management(veg_r, veg_r_proj, dem, "#", "25")

    # get the AOI
    ExtractByMask(veg_r_proj, dem).save(veg_class_r)    

    g_list = arcpy.ListRasters('g_g*')
    for g_file in g_list:
        arcpy.Delete_management(g_file)
        
    if arcpy.Exists(veg_r):
        arcpy.Delete_management(veg_r)
        
    if arcpy.Exists(veg_r_proj):
        arcpy.Delete_management(veg_r_proj)
        
    return veg_class_r
    



[docs]def get_slope_aspect(input_dem):
    """ 
    Calculate the slope and aspect from the input DEM 
     
    :param input_dem: `file` the input DEM 
    
    :return: `file` the reclassified slope
    :return: `file` the reclassified aspect
    """ 
    
    input_folder = os.path.dirname(input_dem) 
    parent = os.path.abspath(pjoin(input_folder, os.pardir))
    output_folder = pjoin(parent, 'output')
    
    arcpy.env.workspace = input_folder

    dem_slope = 'slope'
    dem_aspect = 'aspect'
    aspect_rec = output_folder + '\\aspect_r'
    slope_rec = output_folder + '\\slope_r'

    # Derive slope and aspect ...
    Slope(input_dem, "DEGREE", "1").save(dem_slope)
    Aspect(input_dem).save(dem_aspect)

    # Derive reclassifed aspect...
    Reclassify(dem_aspect, "Value", "-1 0 9;0 22.5 1;22.5 67.5 2;67.5 112.5 3;\
               112.5 157.5 4;157.5 202.5 5;202.5 247.5 6;247.5 292.5 7;\
               292.5 337.5 8;337.5 360 1", "NODATA").save(aspect_rec)

    value_max = arcpy.GetRasterProperties_management(
                dem_slope, "MAXIMUM").getOutput(0)
    remap = "0 0 -6;0 5 1;5 10 8;10 15 15;15 20 22;20 " + \
             str(math.ceil(float(value_max))) + " 29"
    
    log.info('remap the slope ' + remap) 
    
    # Derive reclassifed slope...
    Reclassify(dem_slope, "Value", remap, "NODATA").save(slope_rec)

    g_list = arcpy.ListRasters('g_g*')
    for g_file in g_list:
        arcpy.Delete_management(g_file)        
    
    if arcpy.Exists(dem_slope):
        arcpy.Delete_management(dem_slope)
        
    if arcpy.Exists(dem_aspect):
        arcpy.Delete_management(dem_aspect)  
        
    return slope_rec, aspect_rec




[docs]def get8dire(veg_class, slope, aspect):
    """ 
    Combine the downslope and vegetation for each direction based on table 
    2.4.3 in AS 3959 (2009).
    
    :param veg_class: `file` the classified vegetation
    :param slope: `file` the reclassified slope
    :param aspect: `file` the reclassified aspect
    
    :return: `list` the combined slope and vegetion for 8 directions
    """ 
   
    output_folder = os.path.dirname(veg_class) 
    arcpy.env.workspace = output_folder
    arcpy.env.scratchWorkspace = arcpy.env.workspace    

    dire = ['w', 'e', 'n', 's', 'nw', 'ne', 'se', 'sw']
    
    slope_class_list = []

    for one_dir in dire:                             
        slope_class_one_dire = combine(veg_class, slope, aspect, one_dir)        
        slope_class_list.append(slope_class_one_dire)
            
    if arcpy.Exists(slope):
        arcpy.Delete_management(slope)
        
    if arcpy.Exists(aspect):
        arcpy.Delete_management(aspect)
    
    return slope_class_list
    



[docs]def combine(veg_class, slope, aspect, one_dir):
    """ 
    Combine the downslope and vegetation for a specific direction based on 
    table 2.4.3 in AS 3959 (2009).
    
    :param veg_class: `file` the classified vegetation
    :param slope: `file` the reclassified slope
    :param aspect: `file` the reclassified aspect
    :param one_dir: `string` the specific direction
    
    :return: `file` the combined slope and vegetion for the specific direction
    """ 
    
    output_folder = os.path.dirname(veg_class) 
    arcpy.env.workspace = output_folder
    arcpy.env.scratchWorkspace = arcpy.env.workspace
    
    # extract the downslope cells (aspect is the same as the direction)
    dire_aspect = value_lookup.dire_aspect
    aspect_value = dire_aspect[one_dir]
    
    yes_dire = "Value = " + str(aspect_value)
   
    # Local variables...
    xasp_n = "xasp_n"
    slope_a = "slope_a" 
    slope_class = 'slope_class_' + one_dir + '.img'
      
    ExtractByAttributes(aspect, yes_dire).save(xasp_n) 
    Con(xasp_n, slope, "", "").save(slope_a)

    # sum slope value (-6, 1, 8, 15, 22, 29) with class value (1-7) to get unique
    # value (-5 - 36) representing different slope and class combination
    CellStatistics([slope_a, veg_class], "SUM", "NODATA").save(slope_class)
    
    if arcpy.Exists(xasp_n):
        arcpy.Delete_management(xasp_n)
        
    if arcpy.Exists(slope_a):
        arcpy.Delete_management(slope_a)  
        
    return slope_class




[docs]def bal_calc(vegetation_map, dem):
    """ 
    Calcuate BAL based on vegetation map and DEM.
    
    :param vegetation_map: `file` the original vegetation
    :param dem: `file` the input DEM
    """     
    
    log.info('Reclassify the vegetation map ...   ')
    veg_class = reclass_veg(vegetation_map, dem)
    
    log.info('Derive reclassified slope and reclassified aspect ...   ')
    slope, aspect = get_slope_aspect(dem)
    
    log.info('Combine slope and vegetation ...   ')
    slope_class_list = get8dire(veg_class, slope, aspect)
    
    log.info('Calculate the BAL ...   ')    
    bal_cal(veg_class, slope_class_list)
    
    



[docs]def timer(f):
    """
    Basic timing functions for entire process
    """

    @wraps(f)
    def wrap(*args, **kwargs):
        """
        Wrap
        """

        t1 = time.time()
        res = f(*args, **kwargs)

        tottime = time.time() - t1
        msg = "%02d:%02d:%02d " % \
            reduce(lambda ll, b: divmod(ll[0], b) + ll[1:],
                   [(tottime,), 60, 60])

        log.info("Time for {0}:{1}".format(f.func_name, msg))
        return res

    return wrap

    """
    Create all the necessary output folders.

    :param root: `string` Name of root directory
    :raises OSError: If the directory tree cannot be created.

    """

    output = pjoin(root, 'output')

    if os.path.exists(output):
        shutil.rmtree(output)
    try:
        os.makedirs(output)
    except OSError:
        raise

[docs]def run():
    """
    Run the BAL calculations.

    """

    # add subfolders into path
    cmd_folder = os.path.realpath(
        os.path.abspath(
            os.path.split(
                inspect.getfile(
                    inspect.currentframe()))[0]))
    if cmd_folder not in sys.path:
        sys.path.insert(0, cmd_folder)   

    cmd_subfolder = pjoin(cmd_folder, "utilities")
    if cmd_subfolder not in sys.path:
        sys.path.insert(0, cmd_subfolder)

    config = ConfigParser.RawConfigParser()
    config.read(pjoin(cmd_folder, 'bal_conf.cfg'))

    root = config.get('inputValues', 'root')
    
    logfile = config.get('Logging', 'LogFile')
    logdir = dirname(realpath(logfile))

    # If log file directory does not exist, create it
    if not isdir(logdir):
        try:
            os.makedirs(logdir)
        except OSError:
            logfile = pjoin(os.getcwd(), 'bal.log')

    loglevel = config.get('Logging', 'LogLevel')
    verbose = config.getboolean('Logging', 'Verbose')

    if verbose:
        verbose = True
    else:
        verbose = False

    fl_start_log(logfile, loglevel, verbose)

    # set input maps and output folder
    terrain_map = pjoin(pjoin(root, 'input'), "big_area")
    dem = pjoin(pjoin(root, 'input'), "king_dem")

    do_output_directory_creation(root)
    
    arcpy.env.overwriteOutput = True
    
    arcpy.CheckOutExtension("spatial")
    
    bal_calc(terrain_map, dem)
    

    log.info("Successfully completed BAL calculation")



if __name__ == '__main__':
    run()
