[docs]def bal_cal(veg_class, slope_class_list):
    """ 
    Calculate the BAL based on the classified vegetation and the combined
    slope and vegetation according to the table of determination of bushfire
    attack level (BAL) in AS 3959 (2009).
    
    :param veg_class: `file` the input classified vegetation
    :param slope_class_list: `list` the combined slope and vegetion for 
                              8 directions
    
    :return: `file` the reclassified vegetation 
    """ 
    
    output_folder = os.path.dirname(veg_class) 
    arcpy.env.overwriteOutput = True

    # set directory
    work_folder = output_folder
    os.chdir(work_folder)
    arcpy.env.workspace = work_folder

    # open the image'
    ds = Raster(veg_class)

    if ds is None:
        log.info('Could not open ' + veg_class)
        sys.exit(1)
    
    log.info('The vegetation raster format is %s' % ds.format)
    log.info('The image size is %s' % ds.width + 'x %s' % ds.height) 
    log.info('The image extent is %s' % ds.extent)
    log.info('The image resolution is %s' % ds.meanCellWidth + 
            'x %s' % ds.meanCellHeight) 
    
    veg_data = arcpy.RasterToNumPyArray(ds, nodata_to_value=-99 ) 
    
    dire = ['w', 'e', 'n', 's', 'nw', 'ne', 'se', 'sw']
    
    direction = 0
    
    bal_list = []
    
    for one_dir in dire:

        log.info('The direction is ' + one_dir)

        # open the image
        slope_class = slope_class_list[direction]
        log.info('The combined vege and slope class is ' + slope_class)
        
        ds1 = Raster(slope_class)

        if ds1 is None:
            log.info('Could not open ' + slope_class)
            sys.exit(1)

        # get image size, format, projection
        pixelWidth = ds1.meanCellWidth
        pixelHeight = ds1.meanCellHeight
        
        sr = ds1.spatialReference
        lowleft_corner = arcpy.Point(ds1.extent.XMin,ds1.extent.YMin)
        
        slope_veg_data = arcpy.RasterToNumPyArray(ds1, nodata_to_value=-99 )         
        
        outdata = convo(one_dir, veg_data, slope_veg_data, pixelWidth)
        
        output_dir = 'bal_' + one_dir + '.img' 

        if arcpy.Exists(output_dir):
            arcpy.Delete_management(output_dir)

        arcpy.NumPyArrayToRaster(outdata, lowleft_corner, 
                                 pixelWidth, pixelHeight).save(output_dir) 
        
        arcpy.DefineProjection_management(output_dir, sr)

        del outdata
        del slope_veg_data
        
        if arcpy.Exists(slope_class):
            arcpy.Delete_management(slope_class)
            
        direction += 1
        
        bal_list.append(output_dir)
    
    if arcpy.Exists(veg_class):
        arcpy.Delete_management(veg_class)
              
    CellStatistics(bal_list, "MAXIMUM", "NODATA").save('bal_max.img')
    arcpy.BuildPyramidsandStatistics_management(output_folder, "#", 
                                                "BUILD_PYRAMIDS", 
                                                "CALCULATE_STATISTICS")

    ds = None
    del veg_data  
    
    log.info('finish sucessfully')




[docs]def convo(a_dir, veg_data, slope_veg_data, pixelWidth):
    """ 
    Find the maximum BAL for the point of interest by assessing all 
    neighbours' BAL values in one direction (up to 100 meters).
    
    :param a_dir: `string` the specific direction
    :param veg_data: :class:`numpy.ndarray` the classified vegetation values
    :param slope_veg_data: :class:`numpy.ndarray` the combined vegetation and 
                            slope values
    :param pixelWidth: `float` the pixel width of the combined raster of slope 
                        and vegetation
    
    :return: :class:`numpy.ndarray` the output BAL values
    """ 
    
    # dire_width is the cell span length at the specific direction
    if a_dir in ['w', 'e', 'n', 's']:
        dire_width = pixelWidth
    else:
        dire_width = pixelWidth*1.414

    filter_width = int(numpy.ceil(100/dire_width))    
    log.info('The filter width is ' + str(filter_width))     
    
    rows = veg_data.shape[0]    
    cols = veg_data.shape[1]  
    
    
    outdata = numpy.zeros((rows, cols), numpy.float32)
    outdata.fill(2.0)
    
    for i in range(rows):    
        for jj in range(cols):
            
            # for pixels whose neighbour amount is less than defined value.
            # e.g here for 25m resolution, it is 4 for hori and vert and 3 
            # for diagnoal direction    
            all_neighb_dir = value_lookup.all_neighb[a_dir](i, jj, rows, cols)
            
            if all_neighb_dir < filter_width:
                max_neighb_dir = all_neighb_dir
            else:
                max_neighb_dir = filter_width                
            
            slope_veg = numpy.zeros(max_neighb_dir)
            veg = numpy.zeros(max_neighb_dir)
            bal = numpy.zeros(max_neighb_dir)
            bal.fill(2.0)
            dist = numpy.zeros(max_neighb_dir)
                
            # if max_neighb_dir is 0, it means no neighbours at this direction
            # so no loop is acted as follows. then the output uses the defaut
            # value of 2.
            for m in range(1, max_neighb_dir+1):
                # get neighbour point location
                point_row = value_lookup.point_r[a_dir](i,m)
                point_col = value_lookup.point_c[a_dir](jj,m)                

                # get neighbour point distance, slope_veg, veg data 
                dist[m-1] = (m-1) * dire_width + 0.5 * dire_width
                slope_veg[m-1] = slope_veg_data[point_row, point_col]
                veg[m-1] = veg_data[point_row, point_col]                
                
                # calcualte bal for the neighbour point
                bal[m-1] = bal_esti_80(veg[m-1], dist[m-1], slope_veg[m-1])
            
            
            # get the calculated pixel value
            if len(bal) > 0:                               
                outdata[i, jj] = max(bal)
                if outdata[i, jj] == -99:
                    outdata[i, jj] = 2.0            
          
    return outdata
